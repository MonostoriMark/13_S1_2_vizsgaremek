sudo nano /etc/systemd/system/main.service

[Unit]
Description=Main Application
After=network.target

[Service]
User=pi
WorkingDirectory=/home/pi
ExecStart=/usr/bin/python3 /home/pi/main.py
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target
